apiVersion: apps/v1
kind: Deployment
metadata:
  name: hybrid-ai-app
  namespace: default
spec:
  replicas: 2
  
  # 1. 🔍 [필수 누락 부품] 어떤 라벨을 가진 팟을 감시할지 정의
  selector:
    matchLabels:
      app: hybrid-ai-app

  template:
    metadata:
      # 2. 🔍 [필수 누락 부품] 실제 생성될 팟에 붙일 라벨 정의 (위의 selector와 일치해야 함)
      labels:
        app: hybrid-ai-app
    spec:
      containers:
        - name: hybrid-ai-app
          # 이미지 경로 뒤쪽은 GitLab CI가 자동으로 SHA 태그로 치환하므로 뼈대만 유지하면 됩니다.
          image: 380958218266.dkr.ecr.ap-northeast-2.amazonaws.com/hybrid-ai-app:latest
          ports:
            - containerPort: 5000 # 파이썬 웹 앱 포트에 맞게 설정
