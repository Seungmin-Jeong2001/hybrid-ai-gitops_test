---
- name: EKS 클러스터 위에 ArgoCD 설치 및 환경 구성 자동화
  hosts: localhost
  connection: local
  gather_facts: no

  # vars 섹션의 하드코딩을 제거하고 inventory.ini의 변수를 사용함

  tasks:
    - name: 1. 테라폼이 생성한 EKS 클러스터 접속 정보(Kubeconfig) 갱신 및 로컬 연결
      command: "aws eks update-kubeconfig --region {{ region }} --name {{ cluster_name }}"

    - name: 2. ArgoCD 배포를 위한 쿠버네티스 Namespace 생성
      command: "kubectl create namespace argocd"
      ignore_errors: yes

    - name: 3. EKS 클러스터 내부에 ArgoCD 엔진 공식 매니페스트 설치
      command: "kubectl apply --server-side --force-conflicts -n argocd -f https://raw.githubusercontent.com/argoproj/argo-cd/stable/manifests/install.yaml"

    - name: 4. 외부 브라우저 접속을 위해 ArgoCD 서버를 LoadBalancer 타입으로 변경
      command: "kubectl patch svc argocd-server -n argocd -p '{\"spec\": {\"type\": \"LoadBalancer\"}}'"

    - name: 5. ArgoCD Pod가 준비될 때까지 대기
      command: "kubectl wait --for=condition=Ready pods --all -n argocd --timeout=300s"

    - name: 6. ArgoCD 대시보드 진입용 초기 어드민 비밀번호 추출
      shell: "kubectl -n argocd get secret argocd-initial-admin-secret -o jsonpath='{.data.password}' | base64 -d"
      register: argocd_password
      retries: 10
      delay: 5
      until: argocd_password.rc == 0

    - name: 7. AWS가 발급한 외부 접속용 LoadBalancer DNS 주소 획득
      shell: "kubectl get svc argocd-server -n argocd -o jsonpath='{.status.loadBalancer.ingress[0].hostname}'"
      register: argocd_url
      until: argocd_url.stdout != ""
      retries: 20
      delay: 10

    - name: 8. KServe 의존성 Cert-Manager 설치 (InferenceService 사용 필수)
      command: "kubectl apply -f https://github.com/cert-manager/cert-manager/releases/download/v1.13.0/cert-manager.yaml"

    - name: 9. Cert-Manager 준비 대기
      command: "kubectl wait --for=condition=Ready pods --all -n cert-manager --timeout=300s"

    - name: 10. KServe 설치 (RawDeployment 모드 - 가벼운 설정)
      shell: |
        kubectl apply -f https://github.com/kserve/kserve/releases/download/v0.11.0/kserve.yaml
        kubectl apply -f https://github.com/kserve/kserve/releases/download/v0.11.0/kserve-runtimes.yaml

    - name: 11. KServe 이미지 풀 오류 해결 (kube-rbac-proxy 이미지 교체)
      shell: |
        kubectl set image deployment/kserve-controller-manager -n kserve \
        kube-rbac-proxy=quay.io/brancz/kube-rbac-proxy:v0.13.1

    - name: 12. KServe 기본 배포 모드를 RawDeployment로 변경 (Knative/Istio 미사용 환경)
      shell: |
        kubectl patch configmap inferenceservice-config -n kserve --type merge -p '{"data": {"deploy": "{\"defaultDeploymentMode\": \"RawDeployment\"}"}}'
        kubectl rollout restart deployment/kserve-controller-manager -n kserve

    - name: 13. KServe 컨트롤러 준비 완료 대기
      command: "kubectl wait --for=condition=Available deployment/kserve-controller-manager -n kserve --timeout=300s"

    - name: 14. ArgoCD Image Updater 설치 (Imperative 모드 지원)
      command: "kubectl apply -n argocd -f https://raw.githubusercontent.com/argoproj-labs/argocd-image-updater/stable/manifests/install.yaml"

    - name: 15. ArgoCD Application 및 Image Updater 설정 적용
      command: "kubectl apply -f ../manifest-repo/argocd-app.yaml"

    - name: 🎉 [인프라 구축 완료 보고서]
      debug:
        msg:
          - "=========================================================="
          - " EKS 클러스터: {{ cluster_name }} (VPC: {{ vpc_id }})"
          - " 엔드포인트: {{ cluster_endpoint }}"
          - " ArgoCD URL: http://{{ argocd_url.stdout }}"
          - " ArgoCD Password: {{ argocd_password.stdout }}"
          - "=========================================================="
